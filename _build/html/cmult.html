def PRMAX(Zc):
    X200=np.linspace(0,np.pi,200)
    YCS=CS(Zc.real,Zc.imag)
    Cc=wchiC[-1](Zc)
    PMAX=max(abs(P(X200+1j*YCS(X200),Cc)))
    RMAX=max(abs(R(X200+1j*YCS(X200),Cc)))
    return PMAX,RMAX
