Testing module deepmax#

# global imports
import numpy as np
import matplotlib.pyplot as plt
from scipy.optimize import root

from deepmax.cooker import X_N,cooker,set_n,P,R,Zcooker,Z_N
from deepmax.empirical import Ye
from deepmax.plot import grid,contours

# using mult basis
from deepmax.mult import wchiC
set_n(*wchiC)

N=26
Z0=Z_N(Ye,N,alpha=1.25)
Zc=Zcooker(Z0)
Hc=-Zc[-1].imag
Hc/2/np.pi

np.float64(0.14106343959957984)

# using mult_n basis
from deepmax.mult_n import chi_n,w_n,C_n
set_n(w_n,chi_n,C_n)

N=27
X=X_N(N,alpha=1.25)
Y0=Ye(X)
Yc=cooker(Y0,X)
Hc=-Yc[-1]
Hc/2/np.pi

np.float64(0.14106345768310607)

Z=X+1j*np.array([0,*Yc])
C=C_n(Z)

Grid=grid(0,np.pi-1j)
Z=Grid[0]+1j*Grid[1]

R0=R(0,C)
PatZ=P(Z,C)
RatZ=R(Z,C)
CP=contours(Grid,PatZ,np.linspace(-1,+1,11),'blue')
CR=contours(Grid,RatZ,np.linspace(R0-1,R0+1,11),'orange')
plt.title(f'P=0,R={R0:8.4f}')
plt.show()

X=np.linspace(0,np.pi,200)

def FR(Y):
        Z=X+1j*Y
        return R(Z,C)

def FP(Y):
        Z=X+1j*Y
        return P(Z,C)

YR=root(FR,-np.ones(X.shape)).x
YP=root(FP,-np.ones(X.shape)).x

plt.plot(X,YP)
plt.plot(X,YR)
